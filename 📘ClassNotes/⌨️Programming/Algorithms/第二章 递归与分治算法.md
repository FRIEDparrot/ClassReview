// Algorithm_Basic_Operations.cpp : This file contains the 'main' function. Program execution begins and ends there.
#include <iostream>
#include <string>
#include <stack>
#include <vector>
using namespace std;
int v[100];
void seq(int m, int n) {
    if (m == n) {
        for (int i = 0; i < n; i++) {
            cout << v[i];
        }
        cout << endl;
    }
    else {
        for (int i = m; i < n; i++) {
            if (i == m || v[m] != v[i]) { // when i = m, will continue 
                swap(v[m], v[i]);
                seq(m + 1, n);
                swap(v[m], v[i]);
            }
        }
    }
}

int main()
{
    int n, x;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> x;
        v[i] = x;
    }
    seq(0,n);
    return 0;
}
