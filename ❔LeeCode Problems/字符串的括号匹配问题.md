如果使用栈的方法，可以通过入栈和出栈的操作来进行

```cpp 
class Solution {
	public:
	    bool isValid(string s) {
	        stack<char> sequence;
	        for (int i = 0; i < s.length(); i++) {
	            switch (s[i]) {
	            case '(':
	            case '[':
	            case '{':
	                sequence.push(s[i]);
	                break;
	            case ')':
	                if (sequence.empty()) return false;
	                if (sequence.top() != '(') return false;
	                sequence.pop();
	                break;
	            case ']':
	                if (sequence.empty()) return false;
	                if (sequence.top() != '[') return false;
	                sequence.pop();
	                break;
	            case '}':
	                if (sequence.empty()) return false;
	                if (sequence.top() != '{') return false;
	                sequence.pop();
	                break;
	            }
	        }
	
	        if (sequence.empty()) return true; // 检查是否有栈顶元素, 如果栈为空则
	        return false;
	    }
};
```

另外可以利用哈希表存储对应关系: 
使用哈希表存储对应关系的方式如下：
注意其中的`pairs.count(ch)`存储了对应是否出现

```cpp 
class Solution {
	public:
	    bool isValid(string s) {
	        int n = s.size();
	        if (n % 2 == 1) { //为了保证每个括号匹配，则长度必须是偶数的
	            return false;
	        }
	        unordered_map<char, char> pairs = { 
		        // 建立哈希表存储对应关系
	            {')', '('},
	            {']', '['},
	            {'}', '{'}
	        };
	        stack<char> stk;
	        for (char ch: s) { // 对于每一个字符进行
	            if (pairs.count(ch)) { // 注意哈希表使用count元素计数
		            // 这个元素计数实质上是判断( [ {
	                if (stk.empty() || stk.top() != pairs[ch]) {
	                    return false;
	                }
	                stk.pop();
	            }
	            else {
	                stk.push(ch);
	            }
	        }
	        return stk.empty(); // 最终栈为空empty则为true,否则
	    }
};
```
