$$\begin{matrix}
 \sigma_1 = \sigma_{ij1}^0 \alpha_t \\
 \sigma_2 = \sigma_{ij2}^0 \alpha_t \\
 \sigma_3 = \sigma_{ij3}^0 \alpha_t
 \end{matrix}
$$
